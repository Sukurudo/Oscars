CREATE DATABASE awardwinners
    WITH 
    OWNER = postgres
    ENCODING = 'UTF8'
    LC_COLLATE = 'en_US.UTF-8'
    LC_CTYPE = 'en_US.UTF-8'
    TABLESPACE = pg_default
    CONNECTION LIMIT = -1;

CREATE TABLE movies (
 id INT PRIMARY KEY NOT NULL,
 title TEXT,
 genre TEXT,
 won_awards BOOLEAN,
 rotten_tomatos INT
);
